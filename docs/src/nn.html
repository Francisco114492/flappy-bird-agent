 25class NN:
 26    """
 27    Neural Network (NN) helper class.
 28
 29    It has been devised to be used in a neuroevolution training scheme.
 30    This means that the traditional backpropagation algorithm is not available.
 31    It offers two other methods namely ravel and update.
 32    Ravel reduces the NN to a 1D  vector.
 33    The update method receives a 1D vector and loads the values into a network.
 34    """
 35
 36    def __init__(self, nn_architecture: list) -> None:
 37        """
 38        Constructor for a NN object.
 39
 40        Requires the definition of the network as follows::
 41
 42            NN_ARCHITECTURE = [
 43            {'input_dim': 4, 'output_dim': 2, 'activation': 'relu'},
 44            {'input_dim': 2, 'output_dim': 1, 'activation': 'sigmoid'}
 45            ]
 46
 47        Args:
 48            nn_architecture (list): neural network definition
 49        """
 50        self.nn_architecture = nn_architecture
 51        self.params_values = init_layers(self.nn_architecture)
 52
 53    def predict(self, X: np.ndarray) -> np.ndarray:
 54        """
 55        Implements the forward propagation algorithm for a Neural Network.
 56
 57        Args:
 58            X (np.ndarray): 1D vector that is the input of the network
 59
 60        Returns:
 61            np.ndarray: the output of the network
 62        """
 63        A_curr, _ = full_forward_propagation(
 64            X, self.params_values, self.nn_architecture
 65        )
 66        return A_curr
 67
 68    def predict_activations(self, X: np.ndarray) -> tuple:
 69        """
 70        Implements the forward propagation algorithm for a Neural Network.
 71        It also returns a list with the activations of each node (used for visualization purposes).
 72
 73        Args:
 74            X (np.ndarray): 1D vector that is the input of the network
 75
 76        Returns:
 77            tuple: the output of the network, and the activations of each node
 78        """
 79        A_curr, memory = full_forward_propagation(
 80            X, self.params_values, self.nn_architecture
 81        )
 82        activations = []
 83
 84        for i in range(len(self.nn_architecture)):
 85            activations.append(
 86                compute_activations(
 87                    memory[f"A{i}"], self.nn_architecture[i]["activation"]
 88                )
 89            )
 90        activations.append(
 91            compute_activations(A_curr, self.nn_architecture[-1]["activation"])
 92        )
 93
 94        return A_curr, activations

 96    def ravel(self) -> np.ndarray:
 97        """
 98        Reduces the network into a 1D vector.
 99
100        Returns:
101            np.ndarray: a 1D vector that represents the network
102        """
103        array = []
104        for idx, layer in enumerate(self.nn_architecture):
105            # we number network layers from 1
106            layer_idx = idx + 1
107            # extraction of W for the current layer
108            W_curr = self.params_values["W" + str(layer_idx)]
109            # extraction of b for the current layer
110            b_curr = self.params_values["b" + str(layer_idx)]
111            array.extend(W_curr.ravel())
112            array.extend(b_curr.ravel())
113        return np.array(array)
114
115    def update(self, params: np.ndarray) -> None:
116        """
117        Loads a 1D vector into a network.
118
119        Args:
120            params (np.ndarray): 1D vector that contains the weights and bias of the network
121        """
122        begin = 0
123        for idx, layer in enumerate(self.nn_architecture):
124            # we number network layers from 1
125            layer_idx = idx + 1
126            # extraction of W for the current layer
127            W_curr = self.params_values["W" + str(layer_idx)]
128            n_elems = W_curr.size
129            head = params[begin : begin + n_elems]
130            self.params_values["W" + str(layer_idx)] = head.reshape(W_curr.shape)
131            begin += n_elems
132            # extraction of b for the current layer
133            b_curr = self.params_values["b" + str(layer_idx)]
134            n_elems = b_curr.size
135            head = params[begin : begin + n_elems]
136            self.params_values["b" + str(layer_idx)] = head
137            begin += n_elems
138
139    def layers(self) -> list:
140        """
141        Returns the number of nodes per layer.
142
143        Returns:
144            list: the number of nodes per layer
145        """
146        rv = []
147
148        # get the input
149        rv.append(self.nn_architecture[0]["input_dim"])
150
151        # get the hidden layers
152        for i in range(1, len(self.nn_architecture)):
153            rv.append(self.nn_architecture[1]["input_dim"])
154
155        # get the output
156        rv.append(self.nn_architecture[-1]["output_dim"])
157
158        return rv

172def init_layers(nn_architecture: list, seed: int = 42) -> dict:
173    """
174    Given a network definition it creates and initializes a new network.
175
176    Args:
177        nn_architecture (dict): neural network definition
178        seed (int): seed for the random generator used in the network initialization
179
180    Returns:
181        dict: with the network parameters
182    """
183    rng = np.random.default_rng(seed)
184    params_values = {}
185
186    # iteration over network layers
187    for idx, layer in enumerate(nn_architecture):
188        layer_idx = idx + 1
189        layer_input_size = layer["input_dim"]
190        layer_output_size = layer["output_dim"]
191
192        nin = layer_input_size
193        nout = layer_output_size
194        sd = math.sqrt(6.0 / (nin + nout))
195
196        params_values["W" + str(layer_idx)] = rng.uniform(
197            -sd, sd, (layer_output_size, layer_input_size)
198        )
199        params_values["b" + str(layer_idx)] = np.zeros((layer_output_size, 1))
200
201    return params_values

204def sigmoid(Z: np.ndarray) -> np.ndarray:
205    """
206    Computes the sigmoid function for a 1D vector.
207    $$
208        \\sigma(z) = \\frac{1} {1 + e^{-z}}
209    $$
210
211    Args:
212        Z (np.ndarray): 1D input vector
213
214    Returns:
215        np.ndarray: 1D vector with the results of the sigmoid operation
216    """
217    return np.where(Z >= 0, 1 / (1 + np.exp(-Z)), np.exp(Z) / (1 + np.exp(Z)))

220def sigmoid_backward(dA: np.ndarray, Z: np.ndarray) -> np.ndarray:
221    """
222    Computes the backpropagation step for the sigmoid function.
223
224    Args:
225        dA (np.ndarray): derivative of the previous layer
226        Z (np.ndarray): 1D input vector
227
228    Returns:
229        np.ndarray: 1D vector with the results of the backpropagation step
230    """
231    sig = sigmoid(Z)
232    return dA * sig * (1 - sig)

235def relu(Z: np.ndarray) -> np.ndarray:
236    """
237    Computes the relu function for a 1D vector.
238    $$
239        Relu(z) = max(0, z)
240    $$
241
242    Args:
243        Z (np.ndarray): 1D input vector
244
245    Returns:
246        np.ndarray: 1D vector with the results of the relu operation
247    """
248    return np.maximum(0, Z)

251def relu_backward(dA: np.ndarray, Z: np.ndarray) -> np.ndarray:
252    """
253    Computes the backpropagation step for the relu function.
254
255    Args:
256        dA (np.ndarray): derivative of the previous layer
257        Z (np.ndarray): 1D input vector
258
259    Returns:
260        np.ndarray: 1D vector with the results of the backpropagation step
261    """
262    dZ = np.array(dA, copy=True)
263    dZ[Z <= 0] = 0
264    return dZ

267def swish(Z: np.ndarray) -> np.ndarray:
268    """
269    Computes the swish function for a 1D vector.
270
271    Args:
272        Z (np.ndarray): 1D input vector
273
274    Returns:
275        np.ndarray: 1D vector with the results of the swish operation
276    """
277    return Z * sigmoid(Z)

280def swish_backward(dA: np.ndarray, Z: np.ndarray) -> np.ndarray:
281    """
282    Computes the backpropagation step for the swish function.
283
284    Args:
285        dA (np.ndarray): derivative of the previous layer
286        Z (np.ndarray): 1D input vector
287
288    Returns:
289        np.ndarray: 1D vector with the results of the backpropagation step
290    """
291    sig = sigmoid(Z)
292    return dA * (sig * (1 + Z * (1 - sig)))

295def single_layer_forward_propagation(
296    A_prev: np.ndarray, W_curr: np.ndarray, b_curr: np.ndarray, activation: str = "relu"
297) -> tuple:
298    """
299    Computes a single step of the forward propagation algorithm.
300
301    Args:
302        A_prev (np.ndarray): Result from the previous layer
303        W_curr (np.ndarray): The weights
304        b_curr (np.ndarray): The bias
305        activation (str): the activation function
306
307    Returns:
308        np.ndarray: the forward pass vector
309
310    Raises:
311        Exception: on non-supported activation function
312    """
313    # calculation of the input value for the activation function
314    Z_curr = np.dot(W_curr, A_prev)
315    Z_curr += b_curr
316
317    # selection of activation function
318    if activation == "relu":
319        activation_func = relu
320    elif activation == "sigmoid":
321        activation_func = sigmoid
322    elif activation == "swish":
323        activation_func = swish
324    else:
325        raise Exception("Non-supported activation function")
326
327    # return of calculated activation A and the intermediate Z matrix
328    return activation_func(Z_curr), Z_curr

331def full_forward_propagation(
332    X: np.ndarray, params_values: dict, nn_architecture: list
333) -> tuple:
334    """
335    Computes a single step of the forward propagation algorithm.
336
337    Args:
338        X (np.ndarray): 1D vector that is the input of the network
339        params_values (dict): the network parameters (weights and bias)
340        nn_architecture (dict): neural network definition
341
342    Returns:
343        tuples: network output and the memory dict(for debug)
344
345    Raises:
346        Exception: on non-supported activation function
347    """
348    # creating a temporary memory to store the information needed for a backward step
349    memory = {}
350    # X vector is the activation for layer 0
351    A_curr = X
352
353    # iteration over network layers
354    for idx, layer in enumerate(nn_architecture):
355        # we number network layers from 1
356        layer_idx = idx + 1
357        # transfer the activation from the previous iteration
358        A_prev = A_curr
359
360        # extraction of the activation function for the current layer
361        activ_function_curr = layer["activation"]
362        # extraction of W for the current layer
363        W_curr = params_values["W" + str(layer_idx)]
364        # extraction of b for the current layer
365        b_curr = params_values["b" + str(layer_idx)]
366        # calculation of activation for the current layer
367
368        A_curr, Z_curr = single_layer_forward_propagation(
369            A_prev, W_curr, b_curr, activ_function_curr
370        )
371
372        # saving calculated values in the memory
373        memory["A" + str(idx)] = A_prev
374        memory["Z" + str(layer_idx)] = Z_curr
375
376    # return of prediction vector and a dictionary containing intermediate values
377    return A_curr, memory

380def get_cost_value(Y_hat: np.ndarray, Y: np.ndarray) -> np.ndarray:
381    """
382    Binary crossentropy cost function
383
384    $$
385        J(W,b) = \\frac{1}{m} \\sum{m}{i=1} L(\\hat{y}^{(i)}, y^{(i)});
386        L(\\hat{y},y) = -{(y\\log(\\hat{y}) + (1 - y)\\log(1 - \\hat{y}))}
387    $$
388
389    Args:
390        Y_hat (np.ndarray): the predicted values
391        Y (np.ndarray): the expected values (outcome)
392
393    Returns:
394        np.ndarray: binary crossentropy cost
395    """
396    # number of examples
397    m = Y_hat.shape[1]
398    # calculation of the cost according to the formula
399    cost = -1 / m * (np.dot(Y, np.log(Y_hat).T) + np.dot(1 - Y, np.log(1 - Y_hat).T))
400    return np.squeeze(cost)

403def convert_prob_into_class(probs: np.ndarray) -> np.ndarray:
404    """
405    An auxiliary function that converts probability into class
406
407    Args:
408        probs (np.ndarray): the binary class probabilities
409
410    Returns:
411        np.ndarray: binary classes
412    """
413    probs_ = np.copy(probs)

415    probs_[probs_ <= 0.5] = 0
416    return probs_

434def update(
435    params_values: dict, grads_values: dict, nn_architecture: list, learning_rate: float
436) -> dict:
437    """
438    The goal of this method is to update network parameters using gradient optimisation.
439
440    Args:
441        params_values (dict): the network parameters (weights and bias)
442        grad_values (dict): stores cost function derivatives calculated with respect to these parameters
443        nn_architecture (dict): neural network definition
444        learning_rate (float): the learning rate for the gradient descent optimization method
445
446    Returns:
447        dict: the network parameters
448
449    """
450    # iteration over network layers
451    for layer_idx, layer in enumerate(nn_architecture, 1):
452        params_values["W" + str(layer_idx)] -= (
453            learning_rate * grads_values["dW" + str(layer_idx)]
454        )
455        params_values["b" + str(layer_idx)] -= (
456            learning_rate * grads_values["db" + str(layer_idx)]
457        )
458    return params_values

461def single_layer_backward_propagation(
462    dA_curr: np.ndarray,
463    W_curr: np.ndarray,
464    b_curr: np.ndarray,
465    Z_curr: np.ndarray,
466    A_prev: np.ndarray,
467    activation: str = "relu",
468) -> tuple:
469    """
470    The essence of this algorithm is the recursive use of a chain rule known
471    from differential calculus â calculate a derivative of functions created
472    by assembling other functions, whose derivatives we already know.
473
474    Args:
475        dA_curr (np.ndarray): the derivative of the current activation
476        W_curr (np.ndarray): The weights
477        b_curr (np.ndarray): The bias
478        Z_curr (np.ndarray): the previous values
479        A_prev (np.ndarray): the previous activation values
480        activation (str): the activation function (default relu)
481
482    Returns:
483        tuple: the derivate for the next layer, and the derivatives of the weights and bias
484    """
485    # number of examples
486    m = A_prev.shape[1]
487
488    # selection of activation function
489    if activation == "relu":
490        backward_activation_func = relu_backward
491    elif activation == "sigmoid":
492        backward_activation_func = sigmoid_backward
493    elif activation == "swish":
494        backward_activation_func = swish_backward
495    else:
496        raise Exception("Non-supported activation function")
497
498    # calculation of the activation function derivative
499    dZ_curr = backward_activation_func(dA_curr, Z_curr)
500
501    # derivative of the matrix W
502    dW_curr = np.dot(dZ_curr, A_prev.T) / m
503    # derivative of the vector b
504    db_curr = np.sum(dZ_curr, axis=1, keepdims=True) / m
505    # derivative of the matrix A_prev
506    dA_prev = np.dot(W_curr.T, dZ_curr)
507
508    return dA_prev, dW_curr, db_curr

511def get_cost_derivative(
512    Y_hat: np.ndarray, Y: np.ndarray, cost_function: str = "bce"
513) -> np.ndarray:
514    """
515    Computes the gradient of the cost function with respect to the prediction (Y_hat).
516
517    Args:
518        Y_hat (np.ndarray): The predicted values
519        Y (np.ndarray): The true values
520        cost_function (str): The type of cost function ('bce' or 'mse')
521
522    Returns:
523        np.ndarray: The derivative dA_prev
524    """
525    if cost_function == "bce":
526        # Derivative of Binary Cross Entropy: -(Y/Y_hat) + ((1-Y)/(1-Y_hat))
527        # Adding a small epsilon to avoid division by zero is good practice
528        epsilon = 1e-15
529        Y_hat = np.clip(Y_hat, epsilon, 1 - epsilon)
530        return -(np.divide(Y, Y_hat) - np.divide(1 - Y, 1 - Y_hat))
531
532    elif cost_function == "mse":
533        # Derivative of Mean Squared Error: 2 * (Y_hat - Y) / m
534        # (Note: The /m is often handled in the update step or ignored in simple implementations,
535        # but here is the raw derivative of the sum of squares)
536        return 2 * (Y_hat - Y)
537
538    else:
539        raise Exception(f"Non-supported cost function: {cost_function}")

542def full_backward_propagation(
543    Y_hat: np.ndarray,
544    Y: np.ndarray,
545    memory: dict,
546    params_values: dict,
547    nn_architecture: list,
548) -> dict:
549    """
550    Backward propagation algorithm (for binary classification)
551
552    We start by calculating a derivative of the cost function with
553    respect to the prediction vector â result of forward propagation.
554    Then iterate through the layers of the network starting from the
555    end and calculate the derivatives with respect to all parameters.
556    Ultimately, function returns a python dictionary containing the gradient we are looking for.
557
558    Args:
559        Y_hat (np.ndarray): the predicted values
560        Y (np.ndarray): the expected values (outcome)
561        memory (dict): intermediate values (debug)
562        params_values (dict): the network parameters (weights and bias)
563        nn_architecture (dict): neural network definition
564
565    Returns:
566        dict: stores cost function derivatives calculated with respect to these parameters
567    """
568    grads_values = {}
569
570    # a hack ensuring the same shape of the prediction vector and labels vector
571    Y = Y.reshape(Y_hat.shape)
572
573    # initiation of gradient descent algorithm
574    dA_prev = get_cost_derivative(Y_hat, Y, cost_function="bce")
575
576    for layer_idx_prev, layer in reversed(list(enumerate(nn_architecture))):
577        # we number network layers from 1
578        layer_idx_curr = layer_idx_prev + 1
579        # extraction of the activation function for the current layer
580        activ_function_curr = layer["activation"]
581
582        dA_curr = dA_prev
583
584        A_prev = memory["A" + str(layer_idx_prev)]
585        Z_curr = memory["Z" + str(layer_idx_curr)]
586
587        W_curr = params_values["W" + str(layer_idx_curr)]
588        b_curr = params_values["b" + str(layer_idx_curr)]
589
590        dA_prev, dW_curr, db_curr = single_layer_backward_propagation(
591            dA_curr, W_curr, b_curr, Z_curr, A_prev, activ_function_curr
592        )
593
594        grads_values["dW" + str(layer_idx_curr)] = dW_curr
595        grads_values["db" + str(layer_idx_curr)] = db_curr
596
597    return grads_values

600def train(
601    X: np.ndarray,
602    Y: np.ndarray,
603    nn_architecture: list,
604    epochs: int = 100,
605    learning_rate: float = 0.01,
606    verbose: bool = False,
607    callback: typing.Callable | None = None,
608) -> dict:
609    """
610    The function returns optimized weights obtained as a
611    result of the training and the history of the metrics
612    change during the training.
613
614    Args:
615        X (np.ndarray): 1D vector that is the input of the network
616        Y (np.ndarray): the expected values (outcome)
617        nn_architecture (dict): neural network definition
618        epochs (int): the number of backwards propagation loops (default 100)
619        learning_rate (float): the learning rate for the gradient descent optimization method (default 0.01)
620        verbose (bool): control the output of debug information (default False)
621        callback (typing.Callable): callback function that is called at each epoch (deafult None)
622
623    Returns
624        dict: the network parameters
625    """
626    # initiation of neural net parameters
627    params_values = init_layers(nn_architecture, 42)
628    # initiation of lists storing the history
629    # of metrics calculated during the learning process
630    cost_history = []
631    accuracy_history = []
632
633    # performing calculations for subsequent iterations
634    for i in range(epochs):
635        # step forward
636        Y_hat, cache = full_forward_propagation(X, params_values, nn_architecture)
637
638        # calculating metrics and saving them in history
639        cost = get_cost_value(Y_hat, Y)
640        cost_history.append(cost)
641        accuracy = get_accuracy_value(Y_hat, Y)
642        accuracy_history.append(accuracy)
643
644        # step backward - calculating gradient
645        grads_values = full_backward_propagation(
646            Y_hat, Y, cache, params_values, nn_architecture
647        )
648        # updating model state
649        params_values = update(
650            params_values, grads_values, nn_architecture, learning_rate
651        )
652
653        if i % 50 == 0:
654            if verbose:
655                print(
656                    "Iteration: {:05} - cost: {:.5f} - accuracy: {:.5f}".format(
657                        i, cost, accuracy
658                    )
659                )
660            if callback is not None:
661                callback(i, params_values)
662    return params_values

665def compute_activations(A: np.ndarray, activation: str) -> list:
666    """
667    Check if the layer nodes are active or not (activation functions is high or low)
668
669    Args:
670        A (np.ndarray): the output of a layer
671        activation (str): the activation function
672
673    Returns:
674        list: with 1 (high) for activa and 0 for inactive (low)
675    """
676    if activation == "relu" or activation == "swish":
677        # Returns 1 where A > 0, else 0
678        return list((A > 0).astype(int))
679    elif activation == "sigmoid":
680        # Returns 1 where A >= 0.5, else 0
681        return list((A >= 0.5).astype(int))
682    else:
683        raise Exception("Non-supported activation function")

686def network_size(nn_architecture: list) -> int:
687    """
688    Given a network topology it computnes the number of weigths and bias.
689
690    Args:
691        nn_architecture (dict): neural network definition
692
693    Returns
694        int: the number of weigths and bias
695    """
696    size = 0
697    # iteration over network layers
698    for layer in nn_architecture:
699        # extracting the number of units in layers
700        layer_input_size = layer["input_dim"]

467    A_prev: np.ndarray,
468    activation: str = "relu",
469) -> tuple:
470    """
471    The essence of this algorithm is the recursive use of a chain rule known
472    from differential calculus â calculate a derivative of functions created
473    by assembling other functions, whose derivatives we already know.
474
475    Args:
476        dA_curr (np.ndarray): the derivative of the current activation
477        W_curr (np.ndarray): The weights
478        b_curr (np.ndarray): The bias
479        Z_curr (np.ndarray): the previous values
480        A_prev (np.ndarray): the previous activation values
481        activation (str): the activation function (default relu)
482
483    Returns:
484        tuple: the derivate for the next layer, and the derivatives of the weights and bias
485    """
486    # number of examples
487    m = A_prev.shape[1]
488
489    # selection of activation function
490    if activation == "relu":
491        backward_activation_func = relu_backward
492    elif activation == "sigmoid":
493        backward_activation_func = sigmoid_backward
494    elif activation == "swish":
495        backward_activation_func = swish_backward
496    else:
497        raise Exception("Non-supported activation function")
498
499    # calculation of the activation function derivative
500    dZ_curr = backward_activation_func(dA_curr, Z_curr)
501
502    # derivative of the matrix W
503    dW_curr = np.dot(dZ_curr, A_prev.T) / m
504    # derivative of the vector b
505    db_curr = np.sum(dZ_curr, axis=1, keepdims=True) / m
506    # derivative of the matrix A_prev
507    dA_prev = np.dot(W_curr.T, dZ_curr)
508
509    return dA_prev, dW_curr, db_curr

def get_cost_derivative( Y_hat: numpy.ndarray, Y: numpy.ndarray, cost_function: str = 'bce') -> numpy.ndarray: 
512def get_cost_derivative(
513    Y_hat: np.ndarray, Y: np.ndarray, cost_function: str = "bce"
514) -> np.ndarray:
515    """
516    Computes the gradient of the cost function with respect to the prediction (Y_hat).
517
518    Args:
519        Y_hat (np.ndarray): The predicted values
520        Y (np.ndarray): The true values
521        cost_function (str): The type of cost function ('bce' or 'mse')
522
523    Returns:
524        np.ndarray: The derivative dA_prev
525    """
526    if cost_function == "bce":
527        # Derivative of Binary Cross Entropy: -(Y/Y_hat) + ((1-Y)/(1-Y_hat))
528        # Adding a small epsilon to avoid division by zero is good practice
529        epsilon = 1e-15
530        Y_hat = np.clip(Y_hat, epsilon, 1 - epsilon)
531        return -(np.divide(Y, Y_hat) - np.divide(1 - Y, 1 - Y_hat))
532
533    elif cost_function == "mse":
534        # Derivative of Mean Squared Error: 2 * (Y_hat - Y) / m
535        # (Note: The /m is often handled in the update step or ignored in simple implementations,
536        # but here is the raw derivative of the sum of squares)
537        return 2 * (Y_hat - Y)
538
539    else:
540        raise Exception(f"Non-supported cost function: {cost_function}")
