 41def rect_rect_collision(r1: Rect, r2: Rect) -> bool:
 42    """
 43    Computes the collision of two rectangular shapes.
 44
 45    Args:
 46        r1 (Rect): the first rectangle
 47        r2 (Rect): the second rectangle
 48
 49    Returns:
 50        bool: true in case of collision, false otherwise
 51    """
 52    return (
 53        r1.x < r2.x + r2.w
 54        and r1.x + r1.w > r2.x
 55        and r1.y < r2.y + r2.h
 56        and r1.h + r1.y > r2.y
 57    )

 60class Player:
 61    """
 62    Class that represents a player in the game (a flappy bird).
 63    """
 64
 65    def __init__(self, uuid: str, px: int = 200, py: int = 140) -> None:
 66        """
 67        Constructor for a Player object.
 68
 69        Args:
 70            uuid (str): unique identifier
 71            px (int): initial position x (default 200)
 72            py (int): initial position y (default 140)
 73        """
 74        self.HEIGHT = 42
 75        self.WIDTH = 60
 76        self.px = px
 77        self.py = py
 78        self.v = 0
 79        self.a = 10
 80        self.click = False
 81        self.highscore = 0
 82        self.done = False
 83        self.uuid = uuid

 85    def update(self, dt: float = 1 / 30) -> None:
 86        """
 87        Updates the player position.
 88
 89        Args:
 90            dt (float): the duration of each frame (default 1/30)
 91        """
 92        if self.click:
 93            self.click = False
 94            self.v = -25
 95            self.a = 0
 96        else:
 97            self.a = 10
 98
 99        self.v = self.v + self.a * dt
100        self.py = self.py + self.v * dt

103class Pipe:
104    """
105    Class that represents a pipe in the game.
106    """
107
108    def __init__(self, px: int, py: int) -> None:
109        """
110        Constructor for a Pipe object.
111
112        Args:
113            px (int): initial position x
114            py (int): initial position y
115        """
116        self.HEIGHT = 355
117        self.WIDTH = 60
118        self.GAP = 100
119        self.HEAD = 30
120        self.px = px
121        self.py_top = py
122        self.py_bottom = py + self.GAP
123        self.v = -10
124
125    def update(self, dt: float = 1 / 30) -> None:
126        """
127        Updates the pipe position.
128
129        Args:
130            dt (float): the duration of each frame (default 1/30)
131        """
132        self.px = self.px + self.v * dt
133
134    def collisions(self, players: list, world_height: int) -> set:
135        """
136        Check if any player collided with the pipe.
137
138        Args:
139            players (list): list of active players
140            world_height (int): the world height
141
142        Returns:
143            set: players that collided with the pipe
144        """
145        keys_to_remove = set()
146
147        rect_top = Rect(int(self.px), 0, self.WIDTH, self.py_top)
148        rect_bottom = Rect(
149            int(self.px), self.py_bottom, self.WIDTH, world_height - self.py_bottom
150        )
151
152        for k in players:
153            p = players[k]
154            rect_player = Rect(p.px, p.py, p.WIDTH, p.HEIGHT)
155
156            if rect_rect_collision(rect_player, rect_top) or rect_rect_collision(
157                rect_player, rect_bottom
158            ):
159                keys_to_remove.add(k)
160
161        return keys_to_remove
162
163    def __str__(self):
164        return f"[{self.px} {self.py_top} {self.py_bottom}]"

199        Adds a new player to the world.

210        Returns the current number of active players.
211
212        Returns:
213            int: number of active players

219        Flag the player input (click).

222            ws: websocket that identifies the players

227    def player_neural_network(self, neural_network: dict) -> None:
228        """
229        Neural network definition and activation to be displayed on the viewer.
230
231        Args:
232            neural_network (dict): neural network definition and activation data
233        """
234        if len(self.players) == 1:
235            self.neural_network = neural_network
236
237    def update(self, dt: float = 1 / 30) -> None:
238        """
239        Updates the position of the pipes and the players.
240
241        Args:
242            dt (float): the duration of each frame (default 1/30)
243        """
244        dt *= 6.0
245        [p.update(dt) for p in self.players.values()]
246
247        if self.with_pipes:
248            # generate new pipe
249            if len(self.pipes) < 3:
250                if self.pipes:
251                    previous_pipe = self.pipes[-1]
252                    lower_limit = max(
253                        previous_pipe.HEAD,
254                        previous_pipe.py_top - int(previous_pipe.GAP / 2),
255                    )
256                    upper_limit = min(
257                        self.HEIGHT - previous_pipe.HEAD - previous_pipe.GAP,
258                        previous_pipe.py_top + int(previous_pipe.GAP / 2),
259                    )
260                    py = random.randint(lower_limit, upper_limit)
261                    self.pipes.append(Pipe(previous_pipe.px + 290, py))
262                else:
263                    self.pipes.append(Pipe(self.WIDTH, 150))
264
265            # update the pipes position
266            [p.update(dt) for p in self.pipes]
267
268            # check if first pipe can be removed
269            if self.pipes:
270                if self.pipes[0].px + self.pipes[0].WIDTH <= 0:
271                    self.pipes.pop(0)
272
273    def collisions(self) -> set:
274        """
275        Check if the players have colidded with the pipes or the world
276
277        Returns:
278            set: player to be removed
279        """
280        keys_to_remove = set()
281
282        if self.with_pipes:
283            # collisions with pipes
284            if self.pipes:
285                # all the players have the same px
286                random_player = list(self.players.values())[0]
287                # select pipe
288                closest_pipe = self.pipes[0]
289                closest_distance = self.WIDTH
290                for pipe in self.pipes:
291                    if pipe.px + pipe.WIDTH > random_player.px:
292                        # pipe in front
293                        distance = pipe.px - random_player.px
294                        if distance < closest_distance:
295                            closest_pipe = pipe
296                            closest_distance = distance
297                keys_to_remove.update(
298                    closest_pipe.collisions(self.players, self.HEIGHT)
299                )
300
301        # collisions with world
302        for k in self.players:
303            p = self.players[k]
304            if p.py < 0 or (p.py + p.HEIGHT) > self.HEIGHT:
305                # p.py = 0 if p.py < 0 else self.HEIGHT-p.HEIGHT
306                # p.dead(self.highscore)
307                keys_to_remove.add(k)
308
309        return keys_to_remove
310
311    def update_highscore(self, dt: float = 1 / 30) -> None:
312        """
313        Update the highscore
314
315        Args:
316            dt (float): the duration of each frame (default 1/30)
317        """
318        self.highscore += dt
319
320    def dump(self) -> dict:
321        """
322        Dumps the world content into a dictionary.
323
324        Returns:
325            dict: world content
326        """
327        players = {}
328        for p in self.players.values():
329            players[p.uuid] = {"px": p.px, "py": p.py, "v": p.v, "a": p.a}
330        rv = {
331            "evt": "world_state",
332            "highscore": self.highscore,
333            "generation": self.generation,
334            "players": players,
335            "pipes": [
336                {"px": p.px, "py_t": p.py_top, "py_b": p.py_bottom} for p in self.pipes
337            ],
338        }
339        if self.neural_network:
340            rv["neural_network"] = self.neural_network
341        return rv

350        """
351        Constructor for a GameServer object.

359    async def incomming_handler(self, websocket) -> None:
360        """
361        Manages the incomming messages.
362
363        Args:
364            websocket: websocket
365            path (str): path used by the client
366        """
367        path = websocket.request.path  # Get path from the websocket object
368        try:
369            async for message in websocket:
370                data = json.loads(message)
371                if data["cmd"] == "join":
372                    if path == "/viewer":
373                        self.viewers.add(websocket)
374                    else:
375                        self.world.add_player(websocket, data["id"])
376
377                if data["cmd"] == "click" and path == "/player":
378                    self.world.player_click(websocket)
379
380                if data["cmd"] == "neural_network" and path == "/player":
381                    self.world.player_neural_network(data["neural_network"])
382
383                if data["cmd"] == "training" and path == "/training":
384                    # send the data to the viewers
385                    data["evt"] = "training"
386                    viewers_to_remove = []
387                    for v in self.viewers:
388                        try:
389                            await v.send(json.dumps(data))
390                        except websockets.exceptions.ConnectionClosed:
391                            viewers_to_remove.append(v)
392                    self.viewers.difference_update(viewers_to_remove)
393
394        except websockets.exceptions.ConnectionClosed as c:
395            logger.info(f"Client disconnected: {c}")
396            if websocket in self.viewers:
397                self.viewers.remove(websocket)
398            # TODO: remove players
399            # elif websocket in self.
400
401    async def mainloop(self, args: argparse.Namespace) -> None:
402        """
403        Defines the main loop of the game.
404
405        Args:
406            args (argparse.Namespace): the arguments to configure the game loop
407        """
408        while True:
409            logger.info("Reset game")
410            self.world.reset()
411            logger.info(f"Waiting for {args.n} players")
412            done = False
413            previous = -1
414            while not done:
415                # check if the have all the players
416                if self.world.number_players() != previous:
417                    print(f"Players {self.world.number_players()}")
418                    previous = self.world.number_players()
419                if self.world.number_players() >= args.n:
420                    done = True
421                else:
422                    await asyncio.sleep(1)
423            logger.info("Starting the main loop")
424            done = False
425            while not done:
426                t = time.perf_counter()
427                # Update world state
428                self.world.update(1 / args.f)
429                keys_to_remove = self.world.collisions()
430                for k in keys_to_remove:
431                    await k.send(
432                        json.dumps({"evt": "done", "highscore": self.world.highscore})
433                    )
434                    self.world.players.pop(k)
435                self.world.update_highscore(1 / args.f)
436
437                # Get world state
438                world_state = json.dumps(self.world.dump())
439
440                # share world state wih all players and viewers:
441                viewers_to_remove = []
442                for v in self.viewers:
443                    try:
444                        await v.send(world_state)
445                    except websockets.exceptions.ConnectionClosed:
446                        viewers_to_remove.append(v)
447                self.viewers.difference_update(viewers_to_remove)
448
449                players_to_remove = []
450                for p in self.world.players.keys():
451                    try:
452                        await p.send(world_state)
453                    except websockets.exceptions.ConnectionClosed:
454                        players_to_remove.append(p)
455                [self.world.players.pop(key) for key in players_to_remove]
456
457                # check if the stopping criteria
458                if self.world.number_players() == 0:
459                    done = True
460                elif args.l > 0 and self.world.highscore >= args.l:
461                    done = True
462                    # remove all the players
463                    players_to_remove = []
464                    players_to_remove.extend(self.world.players.keys())
465                    for p in players_to_remove:
466                        await p.send(
467                            json.dumps(
468                                {"evt": "done", "highscore": self.world.highscore}
469                            )
470                        )
471                        self.world.players.pop(p)
472                delay = max(0, 1 / args.f - (time.perf_counter() - t))
473                await asyncio.sleep(delay)
